-- (1) Buscar o nome e ano dos filmes
SELECT Nome, Ano
FROM Filmes

-- (2) Buscar o nome e ano dos filmes, ordenados por ordem crescente pelo ano
SELECT Nome, Ano
FROM Filmes
ORDER BY Ano asc

-- (3) Buscar pelo filme "Deadpool", trazendo o nome, ano e a duração
SELECT Nome, Ano, Duracao
FROM Filmes
WHERE Nome = 'Deadpool'

-- (4) Buscar os filmes lançados em 2016
SELECT Nome, Ano, Duracao
FROM Filmes
WHERE Ano = 2016

-- (5) Buscar os filmes lançados APÓS o ano 2016
SELECT Nome, Ano, Duracao
FROM Filmes
WHERE Ano > 2016

-- (6) Buscar os filmes com a duracao maior que 100 e menor que 150, ordenando pela duracao em ordem crescente
SELECT Nome, Ano, Duracao
FROM Filmes
WHERE Duracao BETWEEN 100 AND 150
ORDER BY Duracao

-- (7) Buscar a quantidade de filmes lançadas no ano, agrupando por ano, ordenando pela duracao em ordem decrescente
SELECT Ano, COUNT(Nome) AS QuantidadeFilmes
FROM Filmes
GROUP BY Ano
ORDER BY QuantidadeFilmes DESC

-- (8) Buscar os Atores do gênero masculino, retornando o PrimeiroNome, UltimoNome
SELECT PrimeiroNome, UltimoNome, Genero
FROM Atores
WHERE Genero = 'M'

-- (9) Buscar os Atores do gênero feminino, retornando o PrimeiroNome, UltimoNome, e ordenando pelo PrimeiroNome
SELECT PrimeiroNome, UltimoNome, Genero
FROM Atores
WHERE Genero = 'F'
ORDER BY PrimeiroNome 

-- (10) Buscar o nome do filme e o gênero
SELECT Filmes.Nome, GROUP_CONCAT(Generos.Genero, ', ') As 'Gêneros do Filme'
FROM Filmes
JOIN FilmeGenero 
ON Filmes.ID = FilmeGenero.IdFilme
JOIN Generos
ON FilmeGenero.IdGenero = Generos.Id
GROUP BY Nome

-- (11) Buscar o nome do filme e o gênero do tipo "Comédia"
SELECT Filmes.Nome, Generos.Genero
FROM Filmes
JOIN FilmeGenero 
ON Filmes.ID = FilmeGenero.IDfilme
JOIN Generos 
ON FilmeGenero.IDgenero = Generos.ID
WHERE Genero = 'Drama';

-- (12) Buscar o nome do filme e os atores, trazendo o PrimeiroNome, UltimoNome e seu Papel
SELECT Filmes.Nome, Atores.PrimeiroNome, Atores.UltimoNome, ElencoFilme.Papel
FROM Filmes
JOIN ElencoFilme
ON Filmes.ID = ElencoFilme.IDfilme
JOIN Atores 

ON ElencoFilme.IDator = Atores.ID;
